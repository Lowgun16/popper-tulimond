# Persistence & Publishing Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make Studio Mode changes permanent — extract MODEL_INVENTORY to its own file, add a save API route, add SAVE/PUBLISH buttons to the sidebar, and activate beforeunload change detection.

**Architecture:** `src/data/inventory.ts` becomes the single source of truth for character data. A dev-only Next.js API route (`POST /api/save-inventory`) receives the current `StudioSlot[]` state, serializes it via `exportInventoryCode()`, and overwrites `inventory.ts` on disk. `StudioInspector` gets `onSave` prop and a state-machine-driven SAVE/PUBLISH UI. `CollectionOverlay` tracks `hasUnsavedChanges` and attaches a `beforeunload` guard.

**Tech Stack:** Next.js 15 App Router, TypeScript, Node.js `fs`/`path` modules, React `useState`/`useEffect`, Tailwind CSS v4.

// src/app/api/save-inventory/route.ts
// Dev-only endpoint — writes studio state to src/data/inventory.ts
import { NextRequest, NextResponse } from "next/server";
import fs from "fs";
import path from "path";
import { exportInventoryCode } from "@/components/studio/studioUtils";
import type { StudioSlot } from "@/components/studio/studioTypes";

const FILE_HEADER = `// src/data/inventory.ts
// ⚠️  This file is auto-generated by Studio Mode. Do not edit manually.
//     Make changes in the Studio, then click SAVE CHANGES.

import type { ShadowConfig } from "@/components/studio/studioTypes";

export interface OutfitItem {
  id: string;
  name: string;
  collection: string;
  colorway: string;
  price: string;
  type: "public" | "vault";
  dotPosition: string;
  lookbook?: { id: string; src: string; label?: string }[];
}

export interface ModelSlot {
  id: string;
  position: string;
  scale: string;
  mobileScale: string;
  zIndex: number;
  imageSrc: string;
  outfit: OutfitItem[];
  shadow?: ShadowConfig;
}

`;

export async function POST(req: NextRequest) {
  // Dev-only guard — safe to deploy to production
  if (process.env.NODE_ENV !== "development") {
    return NextResponse.json({ ok: false, error: "Only available in development" }, { status: 403 });
  }

  let slots: StudioSlot[];
  try {
    const body = await req.json();
    if (!body || !Array.isArray(body.slots)) {
      return NextResponse.json({ ok: false, error: "Missing or invalid slots array" }, { status: 400 });
    }
    slots = body.slots as StudioSlot[];
  } catch {
    return NextResponse.json({ ok: false, error: "Invalid JSON body" }, { status: 400 });
  }

  try {
    const inventoryBody = exportInventoryCode(slots);
    if (!inventoryBody) {
      return NextResponse.json({ ok: false, error: "exportInventoryCode returned empty output" }, { status: 500 });
    }

    const fileContent = FILE_HEADER + inventoryBody + "\n";
    const filePath = path.join(process.cwd(), "src", "data", "inventory.ts");
    fs.writeFileSync(filePath, fileContent, "utf-8");

    return NextResponse.json({ ok: true });
  } catch (e) {
    const message = e instanceof Error ? e.message : String(e);
    return NextResponse.json({ ok: false, error: message }, { status: 500 });
  }
}
```

type SaveState = "idle" | "saving" | "saved" | "error";
type PublishState = "idle" | "publishing";

const [saveState, setSaveState] = React.useState<SaveState>("idle");
const [saveError, setSaveError] = React.useState<string | null>(null);
const [publishState, setPublishState] = React.useState<PublishState>("idle");

const handleSaveClick = async () => {
  if (saveState === "saving") return;
  setSaveState("saving");
  setSaveError(null);
  try {
    await onSave();
    setSaveState("saved");
    setTimeout(() => setSaveState("idle"), 2000);
  } catch (e) {
    const msg = e instanceof Error ? e.message : "Save failed";
    setSaveError(msg);
    setSaveState("error");
    setTimeout(() => { setSaveState("idle"); setSaveError(null); }, 3000);
  }
};

const handlePublishClick = async () => {
  if (publishState === "publishing" || saveState === "saving") return;
  setPublishState("publishing");
  try {
    await onSave(); // Save first
    // TODO: wire to deploy endpoint
    await new Promise((resolve) => setTimeout(resolve, 3000)); // Stub: 3s delay
  } catch {
    // Save error will be surfaced by handleSaveClick — publish silently aborts
  }
  setPublishState("idle");
};
```
